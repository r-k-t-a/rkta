# Switch

Implements switch control

## Examples

### Primary (default)

<Playground>
  {() => {
    const [isOn, setIsOn] = useState(false);
    return <Switch on={isOn} onClick={() => setIsOn(!isOn)} />;
  }}
</Playground>

### Custom color and size

<Playground>
  {() => {
    const [isOn, setIsOn] = useState(false);
    return <Switch color="success" size="3em" on={isOn} onClick={() => setIsOn(!isOn)} />;
  }}
</Playground>
