Select

A fully customizable select control.

See also: Native Select component.

Basic example

<Select
  caption="Select"
  defaultValue={1}
  formatValue={value => `Item ${value}`}
>
  {([isSelected, makeSelectHandler]) => (
    <List>
      <ArrowKeys vertical>
        <ListButton
          bgColor={isSelected(1) && 'color1'}
          onClick={makeSelectHandler(1)}
        >
          Item 1
        </ListButton>
        <ListButton
          bgColor={isSelected(2) && 'color2'}
          onClick={makeSelectHandler(2)}
        >
          Item 2
        </ListButton>
      </ArrowKeys>
    </List>
  )}
</Select>

Controllable

() => {
  const [state, setState] = useState(1)
  return (
    <Select caption="Select" value={state}>
      <List>
        <ArrowKeys vertical>
          <ListButton
            bgColor={state === 1 && 'color1'}
            onClick={() => setState(1)}
          >
            Item 1
          </ListButton>
          <ListButton
            bgColor={state === 2 && 'color2'}
            onClick={() => setState(2)}
          >
            Item 2
          </ListButton>
        </ArrowKeys>
      </List>
    </Select>
  )
}

Customized

<Select
  bgColor="primary"
  caption="Select"
  floatingAreaProps={{ align: 'top-left', blockLevel: false }}
  floatingAreaContentProps={{ rize: 10 }}
  formatValue={value => `Item ${value}`}
  onChange={value => console.log('onChange: ', value)}
  onClose={() => console.log('onClose')}
  onOpen={() => console.log('onOpen')}
  prepend={
    <Addon>
      <Rocket />
    </Addon>
  }
  rize={1}
>
  {([isSelected, makeSelectHandler]) => (
    <List bgColor="primary1">
      <ArrowKeys validTypes={[ListButton]} vertical>
        <ListButton
          bgColor={isSelected(1) && 'color3'}
          onClick={makeSelectHandler(1)}
        >
          Item 1
        </ListButton>
        <ListButton
          bgColor={isSelected(2) && 'color3'}
          onClick={makeSelectHandler(2)}
        >
          Item 2
        </ListButton>
      </ArrowKeys>
    </List>
  )}
</Select>
