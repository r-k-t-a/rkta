# Input

Allows creating both single and multiline inputs with optional suggest popover.

## Import
```javascript
import { Input } from '@rkta/ui';
```

## Basic Example

<Playground>
  <Input caption="Text Input" />
</Playground>


## Suggest Example

<Playground>
  <Input caption="Suggest Input">
    {({ value, setValue }) => {
      const search = option => !value || option.toLowerCase().includes(value.toLowerCase());
      const options = ['London', 'Paris', 'New York'].filter(search);
      const title = options.length ? 'Suggestions' : 'No Suggestions';
      return (
        <List rize={1}>
          <ListTitle>{title}</ListTitle>
          {options.map(option => <ListButton key={option} onClick={() => setValue(option)}>{option}</ListButton>)}
        </List>
      );
    }}
  </Input>
</Playground>
