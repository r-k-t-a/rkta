# Floating Area

Helper component used by `@rkta/ui` to attach stuff to a DOM Elements or a Window.

## Import

```javascript
import { FloatingArea } from '@rkta/ui';
```

## Element mode

<Playground>
  {() => {
    const [active, setActive] = useState(false);
    const [target, setTarget] = useState(null);
    const show = (event) => {
      setTarget(event.target);
      setActive(true);
    };
    return (
      <div style={{ display: 'flex', justifyContent: 'space-between', padding: 8 }}>
        <Button bgColor="primary" onClick={show}>
          Target 1
        </Button>
        <Button bgColor="primary" onClick={show}>
          Target 2
        </Button>
        <Button bgColor="primary" onClick={show}>
          Target 3
        </Button>
        <FloatingArea producer={target} active={active} offset={8}>
          <List rize={1}>
            <ListTitle>Floating Area</ListTitle>
            <ListButton nowrap>Option 1</ListButton>
            <ListButton nowrap>Option 2</ListButton>
            <ListButton nowrap>Option 3</ListButton>
          </List>
        </FloatingArea>
      </div>
    );
  }}
</Playground>

## Window mode

<Playground>
  {() => {
    const [{ active, align }, setState] = useState({ active: false });
    const show = (align) => () => setState({ align, active: true });

    return (
      <div>
        <div style={{ display: 'flex', justifyContent: 'space-between', padding: 8 }}>
          <Button bgColor="primary" onClick={show('top-left')}>
            Top-left
          </Button>

            Top
          </Button>
          <Button bgColor="primary" onClick={show('top-right')}>
            Top-right
          </Button>
        </div>
        <Divider />
        <div style={{ display: 'flex', justifyContent: 'space-between', padding: 8 }}>
          <Button bgColor="primary" onClick={show('left')}>
            Left
          </Button>
          <Button bgColor="primary" onClick={show()}>
            Center
          </Button>
          <Button bgColor="primary" onClick={show('right')}>
            Right
          </Button>
        </div>
        <Divider />
        <div style={{ display: 'flex', justifyContent: 'space-between', padding: 8 }}>
          <Button bgColor="primary" onClick={show('bottom-left')}>

            Bottom
          </Button>
          <Button bgColor="primary" onClick={show('bottom-right')}>
            Bottom-right
          </Button>
        </div>
        <FloatingArea active={active} align={align} offset={80} onHide={hide} blockLevel>
          <List css={{ height: '100%' }} rize={1}>
            <ListTitle>Floating Area</ListTitle>
            <ListButton nowrap>Option 1</ListButton>
            <ListButton nowrap>Option 2</ListButton>
            <ListButton nowrap>Option 3</ListButton>
