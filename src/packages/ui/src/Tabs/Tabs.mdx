# Tabs

Tabs navigation container with animated indication

## Examples

<Playground>
  {() => {
    const [index, setIndex] = useState(0);
    return (
      <Tabs index={index}>
        {[1, 2, 39999].map((key, currentIndex) => (
          <Button key={key} onClick={() => setIndex(currentIndex)}>
            Tab {key}
          </Button>
        ))}
      </Tabs>
    );
  }}
</Playground>
