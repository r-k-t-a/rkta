## Basic example

<Playground>
  <Select caption="Select" defaultValue={1} formatValue={(value) => `Item ${value}`}>
    {([isSelected, makeSelectHandler]) => (
      <List>
        <ArrowKeys vertical>
          <ListButton bgColor={isSelected(1) && 'color1'} onClick={makeSelectHandler(1)}>
            Item 1
          </ListButton>
          <ListButton bgColor={isSelected(2) && 'color2'} onClick={makeSelectHandler(2)}>
            Item 2
          </ListButton>
        </ArrowKeys>
      </List>
    )}
  </Select>
</Playground>

## Controllable

<Playground>
  {() => {
    const [state, setState] = useState(1);
    return (
      <Select caption="Select" value={state}>
        <List>
          <ArrowKeys vertical>
            <ListButton bgColor={state === 1 && 'color1'} onClick={() => setState(1)}>
              Item 1
            </ListButton>
            <ListButton bgColor={state === 2 && 'color2'} onClick={() => setState(2)}>
              Item 2
            </ListButton>
          </ArrowKeys>
        </List>
      </Select>
    );
  }}
</Playground>
