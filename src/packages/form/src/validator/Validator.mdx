import { Playground, Props } from 'docz';
import { useState } from 'react';
import { Form, useErrors } from '../Form';
import { makeValidator } from './makeValidator';
import { Button, Card, CardBody, Divider, Text, Input } from '@rkta/ui';
import flow from 'lodash/flow';

## Validation

Declaratively validate forms against JSON schema.


## Usage
```javascript
import { makeValidator } from '@rkta/form';

const validate = makeValidator(schema);
```


## Example

<Playground>
  {() => {
    const schema = {
      type: 'object',
      properties: {
        login: {
          type: 'string',
          format: 'email',
          messages: {
            required: 'Please enter login',
            format: 'Wrong format'
          }
        },
        password: {
          type: 'string',
          messages: {
            required: 'Please enter password',
          }
        }
      },
      required: ['login', 'password']
    }
    const validate = makeValidator(schema);
    const FieldError = ({ name }) => {
      const [error] = useErrors(name);
      return error ? <Text color="warning" subtitle2 css={{ marginLeft: 16, marginBottom: 8 }}>{error.message}</Text> : null;
    }
    return (
      <Card rize={1} css={{ maxWidth: '450px', margin: '24px auto' }}>
        <CardBody>
          <Form validate={validate} onFormSubmit={flow(Object.entries, alert)}>
            <Input caption="Login" name="login" bgColor="paper2" />
            <FieldError name="login" />
            <Divider invisible />
            <Input caption="Password" name="password" bgColor="paper2" type="password" />
            <FieldError name="password" />
            <Divider invisible />
            <Button bgColor="primary" type="submit" blockLevel>
              Submit
            </Button>
          </Form>
        </CardBody>
      </Card>
    )
  }}
</Playground>

## Arguments

- schema - JSON schema object
- options - [ajv](https://github.com/epoberezkin/ajv) options
